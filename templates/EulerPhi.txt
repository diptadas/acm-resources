/*
   How many numbers less than n are coprime to n :
   phi(n) = n * product_of(1 âˆ’ 1/p)
   120 = 2^3 * 3^1 * 5^1
   phi(120) = 120 * (1 - 1/2) * (1 - 1/3) * (1 - 1/5) = 32
*/

int phi(int n)
{
    int ret=n;
    for (int i=2; i*i<= n; i++)
    {
        if (n%i==0)
        {
            while(n%i==0) n/=i;
            ret-=ret/i;
        }
    }
    if(n>1) ret-=ret/n;
    return ret;
}
